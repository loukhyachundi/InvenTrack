    public void showres() {
        String sql = "SELECT * FROM EMP_IMS";
        
        try{
            stmt = con.createStatement();
            rs = stmt.executeQuery(sql);
            Colnum = 0;
            while (rs.next()) {
                Colnum++;
            }
            String EmpCount = ""+Colnum+"";
            Lemp.setText(EmpCount);
        } 
        catch(Exception e) {
        
        }
    }
    
    public void showdp() {
        String sql = "SELECT * FROM EMP_IMS WHERE ID = '"+ims.Login.loginid+"'";
        
        try{
            stmt = con.createStatement();
            rs = stmt.executeQuery(sql);
            rs.next();
                dpid.setText("ID : "+rs.getString(1));
                dpname.setText("Name : "+rs.getString(3));
                byte[] img = rs.getBytes(12);
                ImageIcon imm = new ImageIcon(img);
                Image im = imm.getImage();
                Image myimg = im.getScaledInstance(dp.getWidth(), dp.getHeight(), Image.SCALE_SMOOTH);
                ImageIcon newimg = new ImageIcon(myimg);
                dp.setIcon(newimg);
        } 
        catch(Exception e) {
        
        }
    }
